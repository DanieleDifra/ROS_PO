4/* Make functionality from the 1993 edition of the POSIX.1b standard available to



5 * get CLOCK_MONOTONIC definition */



6#ifndef _POSIX_C_SOURCE



7#define _POSIX_C_SOURCE 199309L



8#endif



9



10#include <stdio.h>



11#include <sys/time.h>



12#include <time.h>



13



14double ctimefun()



15{



16    double t = 0.0;



17    



18    /* Use CLOCK_MONOTONIC, so that the returned time is not affected by



19     discontinuous jumps in the system time, e.g. on manual time changes or



20     during Daylight Savings Time */



21    



22    struct timespec currentTime;



23    clock_gettime(CLOCK_MONOTONIC, &currentTime);



24    t = (double)currentTime.tv_sec + (double)currentTime.tv_nsec*1e-9;



25    



26    return t;



27}
