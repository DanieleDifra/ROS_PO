97/* Function: DType2Real_T ======================================================



98 * Convert a built-in data type to a real_T value.  Return the real_T value



99 * as well as a string (by reference) corresponding to the original data type.



100 * If the data type is not recognized as a Simulink built-in data type, then



101 * NULL is returned for the dTypeName and 0 is returned for the dTypeValue.



102 */



103PRIVATE double DType2Double(



104    const char              *vPtr,



105    const int               dTypeIdx,



106    const DataTypeTransInfo *dtInfo,



107    const char              **dTypeName)



108{



109    real_T     outVal;



110    char * const *dTypeNames = (char * const *)dtGetDataTypeNames(dtInfo);



111    const char *thisName     = dTypeNames[dTypeIdx];



112



113    *dTypeName = thisName;



114



115    if (strcmp(thisName, "real_T") == 0) {



116        outVal = (real_T) (((const real_T*)vPtr)[0]);



117    } else if (strcmp(thisName, "real32_T") == 0) {



118        outVal = (real_T) (((const real32_T*)vPtr)[0]);



119    } else if (strcmp(thisName, "int8_T") == 0) {



120        outVal = (real_T) (((const int8_T*)vPtr)[0]);



121    } else if (strcmp(thisName, "uint8_T") == 0) {



122        outVal = (real_T) (((const uint8_T*)vPtr)[0]);



123    } else if (strcmp(thisName, "int16_T") == 0) {



124        outVal = (real_T) (((const int16_T*)vPtr)[0]);



125    } else if (strcmp(thisName, "uint16_T") == 0) {



126        outVal = (real_T) (((const uint16_T*)vPtr)[0]);



127    } else if (strcmp(thisName, "int32_T") == 0) {



128        outVal = (real_T) (((const int32_T*)vPtr)[0]);



129    } else if (strcmp(thisName, "uint32_T") == 0) {



130        outVal = (real_T) (((const uint32_T*)vPtr)[0]);



131    } else if (strcmp(thisName, "boolean_T") == 0) {



132        outVal = (real_T) (((const boolean_T*)vPtr)[0]);



133    } else {



134        outVal    = 0;



135        dTypeName = NULL;



136    }



137    return(outVal);



138} /* end DType2Double */

142/* Function: SetParam ==========================================================



143 * Install new parameters.



144 *



145 * NOTE: pbuf looks like:



146 *  [NPARAMS



147 *   B S W DI DATA   % pVal 0



148 *   B S W DI DATA   % pVal 1



149 *   B S W DI DATA   % pVal N



150 *  ]



151 *



152 *  No assumptions about the alignment of pbuf may be made.



153 *



154 *  where,



155 *      B   : Data type transition index.  Note that on the target the data



156 *            type transition index provides both the base address (B)



157 *              of the transition as well as whether or not



158 *              that section of the array contains complex elements.



159 *      S   : starting offset of the data from the DT_TRANS_IDX



160 *      W   : number of elements for this param



161 *      DI  : index into rtw data type table (inUse idx)



162 *      DATA: the param values (in target format).



163 *



164 *  and,



165 *      All values, excluding DATA, are int32_T.



166 */



167#ifndef EXTMODE_DISABLEPARAMETERTUNING



168PUBLIC void SetParam(RTWExtModeInfo  *ei, const char *pbuf)



169{



170    int        i;



171    int32_T    nParams;



172    const char *bufPtr    = pbuf;



173    const int  B          = 0; /* index into dtype tran table (base address)  */



174    const int  SI         = 1; /* starting index - wrt to base address        */



175    const int  W          = 2; /* width of section (number of elements)       */



176    const int  DI         = 3; /* index into data type tables                 */



177    const int  tmpBufSize = sizeof(int32_T) * 4;



178    int32_T    tmpBuf[4];



179



180    const DataTypeTransInfo *dtInfo = (const DataTypeTransInfo *) rteiGetModelMappingInfo(ei);



181    const DataTypeTransitionTable *dtTable = dtGetParamDataTypeTrans(dtInfo);



182    const uint_T *dtSizes = dtGetDataTypeSizes(dtInfo);



183    



184#ifdef MW_DYNAMIC_STRING_SUPPORT



185    const char_T* *dtNames = dtGetDataTypeNames(dtInfo);



186#endif



187



188    /* unpack NPARAMS */



189    (void)memcpy(&nParams, bufPtr, sizeof(int32_T));



190    bufPtr += sizeof(int32_T);



191    



192#ifdef VERBOSE



193    printf("\nUpdating %d parameters....\n", nParams);



194#endif



195



196    /*



197     * Unpack the data and install the new parameters.



198     */



199    for (i=0; i<nParams; i++) {



200        int_T   elSize;



201        int_T   nBytes;



202        char_T  *start;



203        char_T  *tranAddress;



204        int_T   tranIsComplex;



205



206        /* unpack B SI W DI */



207        (void)memcpy(tmpBuf, bufPtr, tmpBufSize);



208        bufPtr += tmpBufSize;



209



210        /* 



211         * Find starting address and size (nBytes) for this parameters 



212         * section of memory.



213         */



214        tranAddress   = dtTransGetAddress(dtTable, tmpBuf[B]);



215        tranIsComplex = dtTransGetComplexFlag(dtTable, tmpBuf[B]);



216



217        elSize = dtSizes[tmpBuf[DI]] * (tranIsComplex ? 2 : 1);



218        nBytes = tmpBuf[W] * elSize;



219



220        /*



221         * String in rapid accelerator is:



222         *   - transferred as character stream, element size is sizeof(char_T)



223         *   - stored/operated as void *, element size is sizeof(void*)



224         * We need to use the later one to calculate "start" address



225         */



226#ifdef MW_DYNAMIC_STRING_SUPPORT



227        if (strcmp(dtNames[tmpBuf[DI]], "string") == 0) {



228            elSize = sizeof(void*);



229        }



230#endif



231        start = tranAddress + (tmpBuf[SI] * elSize);



232



233        /* Install the params. */



234#ifdef MW_DYNAMIC_STRING_SUPPORT



235        if (strcmp(dtNames[tmpBuf[DI]], "string") == 0) {



236            suInitializeString(start, bufPtr);



237            bufPtr += nBytes;



238        } else



239#endif



240        {



241            (void)memcpy(start, bufPtr, nBytes);



242            bufPtr += nBytes;



243        }



244



245#ifdef VERBOSE



246        /*



247         * It is safe to assume that once the params are installed into



248         * the param vector that they are properly aligned.  So we



249         * do our verbosity print-out here.



250         */



251        {



252            double     val;



253            const char *dTypeName;



254            const char *vPtr = (const char *)start;



255



256            val = DType2Double(start, tmpBuf[DI], dtInfo, &dTypeName);



257            printf("\n\tParam| "



258                   "DT_Trans: %d, index: %d, nEls: %d, data type: [%s, %d]\n",



259                   tmpBuf[B], tmpBuf[SI], tmpBuf[W],



260                   (dTypeName != NULL) ? dTypeName : "",tmpBuf[DI]);



261



262            if (!tranIsComplex) {



263                int j;



264                for (j=0; j<tmpBuf[W]; j++) {



265                    val = DType2Double(vPtr, tmpBuf[DI], dtInfo, &dTypeName);



266                    printf("\t\t%f\n", val);



267                    vPtr += dtSizes[tmpBuf[DI]];



268                }



269            } else {



270                int j;



271                for (j=0; j<tmpBuf[W]; j++) {



272                    val = DType2Double(vPtr, tmpBuf[DI], dtInfo, &dTypeName);



273                    printf("\t\t%f + ", val);



274                    vPtr += dtSizes[tmpBuf[DI]];



275



276                    val = DType2Double(vPtr, tmpBuf[DI], dtInfo, &dTypeName);



277                    printf("\t\t%fi\n", val);



278                    vPtr += dtSizes[tmpBuf[DI]];



279                }



280            }



281        }



282#endif



283    }



284} /* end SetParam */



285#endif /* ifndef EXTMODE_DISABLEPARAMETERTUNING */

1477/* Function ====================================================================



1478 * Assign sections in the circular buffer for the requested number of bytes



1479 * (i.e., fill in the bufMem struct).  If there is no room in the circular



1480 * buffer return an overflow error.



1481 *



1482 * NOTE: Do not move the CircularBuffers head forward in this function!  



1483 *       Only move the tmpHead forward.  The actual head is not advanced



1484 *       until the entire time point is successfully copied into the buffer.



1485 *



1486 *       This function modifies tmpHead to point at the next available 



1487 *       location.



1488 *



1489 *       It is possible for tmpHead to equal the tail upon entry to this



1490 *       function.  This does not necessarily mean that the buffer is



1491 *       empty (unwrapped).  It could also mean that the buffer is exactly



1492 *       full (this is considered as wrapped).



1493 */



1494#ifndef EXTMODE_DISABLESIGNALMONITORING



1495PRIVATE boolean_T UploadBufAssignMem(



1496    CircularBuf  *circBuf,



1497    int_T        nBytesToAdd,



1498    char         **tmpHead,   /* in-out */



1499    BufMem       *bufMem)     /* out */



1500{



1501    int_T       nBytesLeft;



1502    boolean_T   overFlow  = false;



1503    char        *end      = circBuf->buf + circBuf->bufSize; /* 1 passed end */



1504



1505    if ((*tmpHead > circBuf->tail) || circBuf->empty) {



1506        /* buffer not wrapped */



1507        nBytesLeft = (int_T)((end - *tmpHead) + (circBuf->tail - circBuf->buf));



1508



1509        if (nBytesLeft < nBytesToAdd) {



1510            overFlow = true;



1511            goto EXIT_POINT;



1512        }



1513



1514        if ((*tmpHead + nBytesToAdd) < end) {



1515            /* still not wrapped */



1516            bufMem->nBytes1  = nBytesToAdd;



1517            bufMem->section1 = *tmpHead;



1518



1519            bufMem->nBytes2  = 0;



1520            bufMem->section2 = NULL;



1521



1522            *tmpHead += nBytesToAdd;



1523        } else {



1524            /* now we're wrapped */



1525            bufMem->nBytes1  = (int_T)(end - *tmpHead);



1526            bufMem->section1 = *tmpHead;



1527



1528            bufMem->nBytes2  = nBytesToAdd - bufMem->nBytes1;



1529            bufMem->section2 = (bufMem->nBytes2 > 0) ? circBuf->buf : NULL;



1530



1531            *tmpHead = circBuf->buf + bufMem->nBytes2;



1532        }  



1533    } else {



1534        /* wrapped */



1535        nBytesLeft = (int_T)(circBuf->tail - *tmpHead);



1536        if (nBytesLeft < nBytesToAdd) {



1537            overFlow = true;



1538            goto EXIT_POINT;



1539        }



1540



1541        bufMem->nBytes1  = nBytesToAdd;



1542        bufMem->section1 = *tmpHead;



1543



1544        bufMem->nBytes2  = 0;



1545        bufMem->section2 = NULL;



1546



1547        *tmpHead += nBytesToAdd;



1548    }



1549    



1550EXIT_POINT:



1551    return(overFlow);



1552} /* end UploadBufAssignMem */

1608/* Function ====================================================================



1609 * If the trigger is in the TRIGGER_FIRED state or we are collecting data for



1610 * pre-triggering, add data, for each tid with a hit, to the upload buffers.  



1611 * This function is called from within the appropriate task, once per sample



1612 * hit.



1613 *



1614 * The format of the packet that is sent to the host is as follows:



1615 *



1616 * definitions:



1617 *      pktType - A qualifier indicating any special action that needs to be



1618 *                taken (e.g., a termination flag following the last data point,



1619 *                or a flag indicating that it is the first data point after



1620 *                a trigger event).



1621 *



1622 *      nBytes - total number of target bytes for this packet (including the



1623 *               nBytes field).  nBytes worth of data represents 1 full time



1624 *               step of the target simulation.



1625 *



1626 *      nSys - The number of systems for which this packet contains data.



1627 *



1628 *      tid - The tid with which this data is associated.



1629 *



1630 *      upInfoIdx - upInfo index



1631 *



1632 *      t - simulation time



1633 *



1634 *      sysId - The index into the BdUploadInfo.sysTables array so that we can



1635 *              map the target data back to the appropriate system.  This is



1636 *              NOT the descendant system index!



1637 *



1638 *      data - the target simulation data (in target format)



1639 *



1640 * The packet looks like:



1641 * [nBytes pktType nSys tid upInfoIdx t sysId [data] sysId [data]...]



1642 *     |                            | |         | |          |



1643 *     ----------------------------- ----------- ------------



1644 *          pkt header          sys data     sys data



1645 *



1646 * Ints are int32_T.



1647 */



1648#ifndef EXTMODE_DISABLESIGNALMONITORING



1649PUBLIC void UploadBufAddTimePoint(int_T tid, real_T taskTime,



1650                                  int32_T upInfoIdx)



1651{



1652    int_T        preTrig;



1653    int_T        overFlow;



1654    TriggerInfo  *trigInfo;



1655    CircularBuf  *circBuf;



1656    BdUploadInfo *uploadInfo = &uploadInfoArray[upInfoIdx];



1657



1658    overFlow   = false;



1659    trigInfo   = &uploadInfo->trigInfo;



1660    circBuf    = &uploadInfo->circBufs[tid];



1661    



1662    /*



1663     * Check for transitions from the TRIGGER_ARMED state to either the



1664     * TRIGGER_FIRED_STATE or the TRIGGER_DELAYED state.  We only do this



1665     * if it is a sample hit for the trigger signal.  Note that this



1666     * is the only place in the whole world that the trigger state can



1667     * move from TRIGGER_ARMED_STATE to TRIGGER_DELAYED or TRIGGER_FIRED.



1668     */



1669    if (trigInfo->state == TRIGGER_ARMED) {



1670        if (trigInfo->trigSignals.nSections == 0) {



1671            /* short-circuit for manual trigger */



1672            trigInfo->state = TRIGGER_FIRED;



1673        } else



1674            if ((tid == trigInfo->tid) &&



1675                (UploadCheckTriggerSignals(upInfoIdx))) {



1676                /* trig signal crossing */



1677                if (trigInfo->delay == 0) {



1678                    trigInfo->state = TRIGGER_FIRED;



1679                    /* 0 unless pre-trig */



1680                    trigInfo->count = trigInfo->preTrig.count;



1681                } else {



1682                    trigInfo->state = TRIGGER_DELAYED;



1683                    assert(trigInfo->count == 0);



1684                    



1685                    /* We will be skipping this step, so the delay count is 1. */



1686                    trigInfo->count = 1;



1687                }



1688            }



1689    }



1690    



1691    preTrig = (trigInfo->state == TRIGGER_ARMED) &&



1692        (trigInfo->preTrig.duration > 0);



1693    



1694    /*



1695     * Handle adding data to the collection buffers - if needed.



1696     */



1697    if (((trigInfo->state == TRIGGER_FIRED) || preTrig) &&



1698        /* bufSize == 0 means no signals in this tid */



1699        circBuf->bufSize != 0) {



1700        



1701        int32_T     i;



1702        BufMem      bufMem;



1703        BufMem      pktStart;



1704        int_T       size;



1705        char_T *tmpHead    = circBuf->head;



1706        const int_T PKT_TYPE_IDX = 0;



1707        const int_T NBYTES_IDX   = 1;



1708        const int_T NSYS_IDX     = 2;



1709        const int_T TID_IDX      = 3;



1710        const int_T UPINFO_IDX   = 4;



1711



1712        int32_T intHdr[5] = {0, 0, 0, 0, 0};



1713        intHdr[UPINFO_IDX] = upInfoIdx;



1714        



1715        if (preTrig && (trigInfo->preTrig.count==trigInfo->preTrig.duration)) {



1716            /* Advance the tail (we don't need the oldest point anymore). */



1717            char *end = circBuf->buf + circBuf->bufSize;



1718            MOVE_TAIL_ONESTEP(circBuf, end);



1719            trigInfo->preTrig.count--;



1720        }



1721        



1722        /*



1723         * Save some space for the 5 integer values that make up the packet



1724         * header: [pktType nBytes nSys tid upInfoIdx].



1725         * The values are filled in later.



1726         */



1727        size = 5*sizeof(int32_T);



1728        overFlow = UploadBufAssignMem(circBuf, size, &tmpHead, &pktStart);



1729        if (overFlow) goto EXIT_POINT;



1730



1731        /*



1732         * We do not want to include the packet type and number of bytes



1733         * in the size calculation.  Size should represent the payload of



1734         * this packet.  The packet type and number of bytes represent the



1735         * packet header and are not included in the payload size.



1736         */



1737        size -= 2*sizeof(int32_T);



1738        intHdr[NBYTES_IDX] += size;



1739        



1740        /* time */



1741        overFlow =



1742            UploadBufAssignMem(circBuf, sizeof(real_T), &tmpHead, &bufMem);



1743        if (overFlow) goto EXIT_POINT;



1744        intHdr[NBYTES_IDX] += sizeof(real_T);



1745        



1746        CIRCBUF_COPY_DATA(bufMem, &taskTime);



1747



1748        /*



1749         * Check each system for an UploadMap. 



1750         */



1751        for (i=0; i<uploadInfo->nSys; i++) {



1752            const SysUploadTable *sysTable =



1753                (const SysUploadTable *)&uploadInfo->sysTables[i];



1754            



1755            if ( (*sysTable->enableState != SUBSYS_RAN_BC_DISABLE) && 



1756                 (*sysTable->enableState != SUBSYS_RAN_BC_ENABLE_TO_DISABLE) ) {



1757                UploadMap *map = sysTable->uploadMap[tid];



1758



1759                if (map != NULL) {



1760                    int_T section;



1761                    intHdr[NSYS_IDX]++;



1762                    



1763                    /* Add system index */



1764                    size = sizeof(int32_T);



1765                    overFlow =



1766                        UploadBufAssignMem(circBuf, size, &tmpHead, &bufMem);



1767                    if (overFlow) goto EXIT_POINT;



1768                    intHdr[NBYTES_IDX] += size;



1769                    



1770                    CIRCBUF_COPY_DATA(bufMem, &i);



1771                    



1772                    /* Add data values */



1773                    for (section=0; section<map->nSections; section++) {



1774                        UploadSection *sect = &map->sections[section];



1775                        



1776#ifdef MW_DYNAMIC_STRING_SUPPORT



1777                        if (sect->isString) {



1778                            /* String data is [nBytes "abc...\0"] */



1779                            const void* strPtr = *(void**)sect->start;



1780                            /* Add number of bytes of the string, including null-ternimator */



1781                            int32_T strNBytes = suStrlen(strPtr) + 1;



1782                            char *tmpStr;



1783                            



1784                            overFlow = UploadBufAssignMem(circBuf, size, &tmpHead, &bufMem);



1785                            if (overFlow) goto EXIT_POINT;



1786                            intHdr[NBYTES_IDX] += size;



1787                            CIRCBUF_COPY_DATA(bufMem, &strNBytes);



1788



1789                            /* Add character bytes */



1790                            overFlow = UploadBufAssignMem(circBuf, strNBytes, &tmpHead, &bufMem);



1791                            if (overFlow) goto EXIT_POINT;



1792                            intHdr[NBYTES_IDX] += strNBytes;



1793                            



1794                            tmpStr = suToCStr(strPtr);



1795                            CIRCBUF_COPY_DATA(bufMem, tmpStr);



1796                            free(tmpStr);



1797                        } else



1798#endif



1799                        



1800                        {



1801                            /* Regular cases */



1802                            overFlow = UploadBufAssignMem(



1803                                circBuf, sect->nBytes, &tmpHead, &bufMem);



1804                            if (overFlow) goto EXIT_POINT;



1805                            intHdr[NBYTES_IDX] += sect->nBytes;



1806                            



1807                            CIRCBUF_COPY_DATA(bufMem, sect->start);



1808                        }



1809                    }



1810                }



1811            }



1812        }



1813



1814        /* If no systems were active then, do nothing. */



1815        if (intHdr[NSYS_IDX] == 0) goto EXIT_POINT;



1816        



1817        /*



1818         * Go back and finish the header: [nBytes pktType nSys tid]



1819         */



1820        



1821        /* ...pktType */



1822        intHdr[PKT_TYPE_IDX] = EXT_UPLOAD_LOGGING_DATA;



1823



1824        /* ...tid */



1825        intHdr[TID_IDX] = tid;



1826        CIRCBUF_COPY_DATA(pktStart, intHdr);



1827



1828        /*



1829         * Time point successfully added to queue.



1830         */



1831        circBuf->head  = tmpHead;



1832        circBuf->empty = false;



1833        



1834        if (preTrig) {



1835            trigInfo->preTrig.count++;



1836        }



1837    }



1838



1839EXIT_POINT:



1840    if (!preTrig) {



1841        if (overFlow) {



1842            trigInfo->overFlow = true;



1843            trigInfo->state    = TRIGGER_TERMINATING;



1844        }



1845#ifdef VXWORKS



1846        else if (trigInfo->state == TRIGGER_FIRED) {



1847            /* allow upload server to run - if data needs to be uploaded */



1848            semGive(uploadSem);



1849        }



1850#endif



1851    } 



1852} /* end UploadBufAddTimePoint */
